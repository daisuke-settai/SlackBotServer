CREATE DATABASE residential_management;
CREATE TABLE residential_management.user (
  id INT NOT NULL AUTO_INCREMENT,
  name VARCHAR(256) NOT NULL UNIQUE,
  enable BOOLEAN NOT NULL DEFAULT TRUE,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id)
);

CREATE TABLE residential_management.macAddress (
  id INT NOT NULL AUTO_INCREMENT,
  mac_address CHAR(17) NOT NULL UNIQUE, # MACアドレスの共用は不可
  user_id INT NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  FOREIGN KEY (user_id) REFERENCES user(id)
);

CREATE TABLE residential_management.logData (
  id INT NOT NULL AUTO_INCREMENT,
  user_id INT NOT NULL,
  log_time TIMESTAMP NOT NULL, # 1時間スパンで記録
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  FOREIGN KEY (user_id) REFERENCES user(id)
);
